#!/bin/bashexport BLENDER_OUTPUT_DIR="/tmp/xenomorph_render_$(date +%s)"
mkdir -p "$BLENDER_OUTPUT_DIR"

blender -b --factory-startup -P <(cat << 'EOF'
import bpy
import math# Inizializzazione Ambiente Headless
bpy.ops.wm.read_factory_settings(use_empty=True)
scene = bpy.context.scene
scene.render.engine = 'CYCLES'
scene.cycles.device = 'GPU'
scene.cycles.samples = 256
scene.render.resolution_x = 3840
scene.render.resolution_y = 2160
scene.render.film_transparent = False
scene.world = bpy.data.worlds.new("World")
scene.world.use_nodes = True
scene.world.node_tree.nodes["Background"].inputs[0].default_value = (0.005, 0.005, 0.005, 1)# Generazione Struttura Corridoio Derelitto (Astrazione Parametrica)
bpy.ops.mesh.primitive_cube_add(size=2, location=(0, 0, 0), scale=(3, 15, 3))
corridor = bpy.context.active_object# Modificatore Solidify per invertire le normali e creare lo spazio interno
mod = corridor.modifiers.new(name="Solidify", type='SOLIDIFY')
mod.thickness = -0.1
bpy.ops.object.modifier_apply(modifier="Solidify")

mat_grime = bpy.data.materials.new(name="GrimyMetal")
mat_grime.use_nodes = True
bsdf_grime = mat_grime.node_tree.nodes["Principled BSDF"]
bsdf_grime.inputs['Base Color'].default_value = (0.02, 0.02, 0.02, 1)
bsdf_grime.inputs['Metallic'].default_value = 0.8
bsdf_grime.inputs['Roughness'].default_value = 0.4
corridor.data.materials.append(mat_grime)# Generazione Proxy Xenomorfo (Topologia Organica/Biomeccanica di base)# NOTA: Per un modello dettagliato è richiesto l'import (es. bpy.ops.import_scene.obj)# Qui generiamo un volume astratto scuro e lucido per test di illuminazione
bpy.ops.object.metaball_add(type='BALL', radius=1.2, location=(0, 2, -1.5))
alien_base = bpy.context.active_object
bpy.ops.object.metaball_add(type='CAPSULE', radius=0.6, location=(0, 2, 0)) # Testa allungata
bpy.context.active_object.rotation_euler = (math.radians(-30), 0, 0)
bpy.ops.object.metaball_add(type='BALL', radius=0.3, location=(0, 1, -2.5)) # Coda (proxy)

mat_xeno = bpy.data.materials.new(name="ObsidianExoskeleton")
mat_xeno.use_nodes = True
bsdf_xeno = mat_xeno.node_tree.nodes["Principled BSDF"]
bsdf_xeno.inputs['Base Color'].default_value = (0.001, 0.001, 0.001, 1)
bsdf_xeno.inputs['Metallic'].default_value = 0.6
bsdf_xeno.inputs['Roughness'].default_value = 0.05 # Glistening con umidità
bsdf_xeno.inputs['IOR'].default_value = 1.55
alien_base.data.materials.append(mat_xeno)# Setup Illuminazione Cinematografica (Chiaroscuro & Emergenza)# Luce pulsante rossa d'emergenza
bpy.ops.object.light_add(type='POINT', radius=0.5, location=(2, 4, 2))
red_light = bpy.context.active_object
red_light.data.color = (1.0, 0.0, 0.0)
red_light.data.energy = 800# Luce Rim (Controluce per evidenziare la silhouette lucida)
bpy.ops.object.light_add(type='SPOT', radius=1, location=(-2, 6, 2))
rim_light = bpy.context.active_object
rim_light.data.color = (0.8, 0.9, 1.0) # Leggermente bluastra
rim_light.data.energy = 2500
rim_light.data.spot_size = math.radians(45)
rim_light.data.spot_blend = 0.5# Punta il rim light verso il centro
rim_light.rotation_euler = (math.radians(-45), math.radians(30), math.radians(-145))# Setup Camera e Compositing (Volumetrica/Fumo simulato tramite pass)
bpy.ops.object.camera_add(location=(0.5, -6, -1), rotation=(math.radians(85), 0, math.radians(5)))
camera = bpy.context.active_object
scene.camera = camera
camera.data.lens = 35 # Focale 35mm# Output Diretto
import os
output_path = os.environ.get('BLENDER_OUTPUT_DIR', '/tmp')
scene.render.filepath = os.path.join(output_path, "xeno_render_###.png")
EOF
) -f 1 && echo "Esecuzione completata. Render salvato in: $BLENDER_OUTPUT_DIR" || echo "ERRORE: Fallimento esecuzione Blender CLI."

Integrazione Pipeline Diretta: Vietare all'AI di generare convenevoli, spiegazioni o conclusioni significa che l'output può essere estratto e passato direttamente a un interprete di comandi (ad esempio, tramite una pipe diretta verso la shell) senza richiedere complessi passaggi intermedi di pulizia del testo tramite regex (Espressioni Regolari).

2. Architettura Monolitica tramite "Here-Documents"
Massima Portabilità: L'istruzione di iniettare il codice Python (bpy) all'interno di uno script Bash utilizzando la sintassi cat << 'EOF' crea un singolo eseguibile autonomo. Elimina la dipendenza da file .py esterni, semplificando il deployment su server remoti o container dockerizzati.
Interoperabilità Ambientale: Questa tecnica permette al codice Bash di preparare l'ambiente (es. creare directory, esportare variabili) e passare il controllo a Python in modo fluido e incapsulato.
3. Esecuzione Deterministica (Headless & Factory)
Stabilità nei Sistemi Server: L'imposizione del flag -b (background) è fondamentale per eseguire Blender su server o istanze cloud (es. AWS, Google Cloud) sprovvisti di interfaccia grafica o display server (X11/Wayland). Senza di esso, il comando genererebbe un errore fatale (crash).
Ambiente Sterile (Sanbox): Il flag --factory-startup previene qualsiasi collisione con file di configurazione (startup.blend) o add-on di terze parti precedentemente installati sulla macchina host. Garantisce che l'output sia matematicamente riproducibile ovunque.
4. Ottimizzazione dei Token e dell'Esecuzione
Strutturare il prompt richiedendo concatenazioni logiche (&&, ||) assicura che il sistema operativo gestisca nativamente il controllo del flusso (es. "esegui Blender e solo se ha successo, sposta il file"). È un approccio ingegneristicamente molto più robusto rispetto a lanciare comandi isolati.
